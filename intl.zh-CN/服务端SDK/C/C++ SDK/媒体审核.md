    #include <string>
    #include <map>
    #include <jsoncpp/json/json.h>
    #include "vod_sdk/openApiUtil.h"

     * 构建审核内容
     */

            Json::Value auditContents;
            Json::Value auditContent;
            auditContent["VideoId"] = "3ebc10160bda481ca9b6858a0b32****"; // 视频ID
            auditContent["Status"] = "Blocked"; // 审核状态
